072        /**
073         * Left-justify a block of text.  Line breaks are preserved, but long lines
074         * are broken.
075         *
076         * @param pw where to output the formatted text
077         * @param text the text to be written
078         * @param linePrefix a string to prepend to each output line
079         * @param lineSuffix a string to append to each output line
080         * @param maxTextPerLine the maximum number of characters to place on
081         *        each line, not counting the prefix and suffix.  If this is -1,
082         *        never break lines.
083         **/
084        static void fillText(
085            PrintWriter pw, String text, String linePrefix, String lineSuffix,
086            int maxTextPerLine)
087        {
088            int i = 0;
089            for (;;) {
090                int end = text.length();
091                if (end <= i) {
092                    // Nothing left.  We're done.
093                    break;
094                }
095    
096                if (i > 0) {
097                    // End the previous line and start another.
098                    pw.println(lineSuffix);
099                    pw.print(linePrefix);
100                }
101    
102                int nextCR = text.indexOf("\r", i);
103                if (nextCR >= 0 && nextCR < end) {
104                    end = nextCR;
105                }
106                int nextLF = text.indexOf("\n", i);
107                if (nextLF >= 0 && nextLF < end) {
108                    end = nextLF;
109                }
110    
111                if (maxTextPerLine > 0 && i + maxTextPerLine <= end) {
112                    // More than a line left.  Break at the last space before the
113                    // line limit.
114                    end = text.lastIndexOf(" ",i + maxTextPerLine);
115                    if (end < i) {
116                        // No space exists before the line limit; look beyond it.
117                        end = text.indexOf(" ",i);
118                        if (end < 0) {
119                            // No space anywhere in the line.  Take the whole line.
120                            end = text.length();
121                        }
122                    }
123                }
124    
125                pw.print(text.substring(i, end));
126    
127                // The line is short enough.  Print it, and find where the next one
128                // starts.
129                i = end;
130                while (i < text.length() &&
131                       (text.charAt(i) == ' ' ||
132                        text.charAt(i) == '\r' ||
133                        text.charAt(i) == '\n')) {
134                    i++;
135                }
136            }
137        }

149        /**
150         * Creates a file-protocol URL for the given filename.
151         **/
152        static URL convertPathToURL(File file)
153        {
154            try {
155                String path = file.getAbsolutePath();
156                // This is a bunch of weird code that is required to
157                // make a valid URL on the Windows platform, due
158                // to inconsistencies in what getAbsolutePath returns.
159                String fs = System.getProperty("file.separator");
160                if (fs.length() == 1)
161                {
162                    char sep = fs.charAt(0);
163                    if (sep != '/')
164                        path = path.replace(sep, '/');
165                    if (path.charAt(0) != '/')
166                        path = '/' + path;
167                }
168                path = "file://" + path;
169                return new URL(path);
170            } catch (MalformedURLException e) {
171                throw new java.lang.Error(e.getMessage());
172            }
173        }

175        static String formatError(String template, Object[] args)
176        {
177            String s = template;
178            for (int i = 0; i < args.length; i++) {
179                String arg = args[i].toString();
180                s = replace(s, "%" + (i + 1), arg);
181                s = replace(s, "%i" + (i + 1), arg);
182            }
183            return s;
184        }

186        /** Returns <code>s</code> with every instance of <code>find</code>
187         * converted to <code>replace</code>. */
188        static String replace(String s,String find,String replace) {
189            // let's be optimistic
190            int found = s.indexOf(find);
191            if (found == -1) {
192                return s;
193            }
194            StringBuffer sb = new StringBuffer(s.length());
195            int start = 0;
196            for (;;) {
197                for (; start < found; start++) {
198                    sb.append(s.charAt(start));
199                }
200                if (found == s.length()) {
201                    break;
202                }
203                sb.append(replace);
204                start += find.length();
205                found = s.indexOf(find,start);
206                if (found == -1) {
207                    found = s.length();
208                }
209            }
210            return sb.toString();
211        }

213        /** Return <code>val</code> in double-quotes, suitable as a string in a
214         * Java or JScript program.
215         *
216         * @param val the value
217         * @param nullMeansNull whether to print a null value as <code>null</code>
218         *   (the default), as opposed to <code>""</code>
219         */
220        static String quoteForJava(String val,boolean nullMeansNull)
221        {
222            if (val == null) {
223                return nullMeansNull ? "null" : "";
224            }
225            String s0;
226            s0 = replace(val, "\\", "\\\\");
227            s0 = replace(val, "\"", "\\\"");
228            s0 = replace(s0, "\n\r", "\\n");
229            s0 = replace(s0, "\n", "\\n");
230            s0 = replace(s0, "\r", "\\r");
231            return "\"" + s0 + "\"";
232        }
233    
234        static String quoteForJava(String val)
235        {
236            return quoteForJava(val,true);
237        }

240         * Returns a string quoted so that it can appear in a resource file.
241         */
242        static String quoteForProperties(String val) {
243            String s0;
244            s0 = replace(val, "\\", "\\\\");
245    //      s0 = replace(val, "\"", "\\\"");
246    //      s0 = replace(s0, "'", "''");
247            s0 = replace(s0, "\n\r", "\\n");
248            s0 = replace(s0, "\n", "\\n");
249            s0 = replace(s0, "\r", "\\r");
250            s0 = replace(s0, "\t", "\\t");
251            return s0;
252        }

256        static String fileNameToClassName(String fileName, String suffix) {
257            String s = fileName;
258            s = removeSuffix(s, suffix);
259            s = s.replace(fileSep, '.');
260            s = s.replace('/', '.');
261            int score = s.indexOf('_');
262            if (score >= 0) {
263                s = s.substring(0,score);
264            }
265            return s;

284        static String removeSuffix(String s, final String suffix) {
285            if (s.endsWith(suffix)) {
286                s = s.substring(0,s.length()-suffix.length());
287            }
288            return s;

291        /**
292         * Given <code>happy/BirthdayResource_en_US.xml</code>,
293         * returns the locale "en_US".
294         */
295        static Locale fileNameToLocale(String fileName, String suffix) {
296            String s = removeSuffix(fileName, suffix);
297            int score = s.indexOf('_');
298            if (score <= 0) {
299                return null;
300            } else {
301                String localeName = s.substring(score + 1);
302                return parseLocale(localeName);
303            }
304        }
305    
306        /**
307         * Parses 'localeName' into a locale.
308         */
309        static Locale parseLocale(String localeName) {
310            int score1 = localeName.indexOf('_');
311            String language, country = "", variant = "";
312            if (score1 < 0) {
313                language = localeName;
314            } else {
315                language = localeName.substring(0, score1);
316                if (language.length() != 2) {
317                    return null;
318                }
319                int score2 = localeName.indexOf('_',score1 + 1);
320                if (score2 < 0) {
321                    country = localeName.substring(score1 + 1);
322                    if (country.length() != 2) {
323                        return null;
324                    }
325                } else {
326                    country = localeName.substring(score1 + 1, score2);
327                    if (country.length() != 2) {
328                        return null;
329                    }
330                    variant = localeName.substring(score2 + 1);
331                }
332            }
333            return new Locale(language,country,variant);
334        }

336        /**
337         * Given "happy/BirthdayResource_fr_FR.properties" and ".properties",
338         * returns "happy/BirthdayResource".
339         */
340        static String fileNameSansLocale(String fileName, String suffix) {
341            String s = removeSuffix(fileName, suffix);
342            // If there are directory names, start reading after the last one.
343            int from = s.lastIndexOf(fileSep);
344            if (from < 0) {
345                from = 0;
346            }
347            while (from < s.length()) {
348                // See whether the rest of the filename after the current
349                // underscore is a valid locale name. If it is, return the
350                // segment of the filename before the current underscore.
351                int score = s.indexOf('_', from);
352                Locale locale = parseLocale(s.substring(score+1));
353                if (locale != null) {
354                    return s.substring(0,score);
355                }
356                from = score + 1;
357            }
358            return s;
359        }

361        /**
362         * Converts a chain of {@link Throwable}s into an array.
363         **/
364        static Throwable[] toArray(Throwable err)
365        {
366            ArrayList list = new ArrayList();
367            while (err != null) {
368                list.add(err);
369                err = getCause(err);
370            }
371            return (Throwable[]) list.toArray(emptyThrowableArray);
372        }
373    
374        private static final Class[] emptyClassArray = new Class[0];
375    
376        private static Throwable getCause(Throwable err) {
377            if (err instanceof InvocationTargetException) {
378                return ((InvocationTargetException) err).getTargetException();
379            }
380            try {
381                Method method = err.getClass().getMethod(
382                        "getCause", emptyClassArray);
383                if (Throwable.class.isAssignableFrom(method.getReturnType())) {
384                    return (Throwable) method.invoke(err, new Object[0]);
385                }
386            } catch (NoSuchMethodException e) {
387            } catch (SecurityException e) {
388            } catch (IllegalAccessException e) {
389            } catch (IllegalArgumentException e) {
390            } catch (InvocationTargetException e) {
391            }
392            try {
393                Method method = err.getClass().getMethod(
394                        "getNestedThrowable", emptyClassArray);
395                if (Throwable.class.isAssignableFrom(method.getReturnType())) {
396                    return (Throwable) method.invoke(err, new Object[0]);
397                }
398            } catch (NoSuchMethodException e) {
399            } catch (SecurityException e) {
400            } catch (IllegalAccessException e) {
401            } catch (IllegalArgumentException e) {
402            } catch (InvocationTargetException e) {
403            }
404            return null;
405        }

407        /**
408         * Formats an error, which may have chained errors, as a string.
409         */
410        static String toString(Throwable err)
411        {
412            StringWriter sw = new StringWriter();
413            PrintWriter pw = new PrintWriter(sw);
414            Throwable[] throwables = toArray(err);
415            for (int i = 0; i < throwables.length; i++) {
416                Throwable throwable = throwables[i];
417                if (i > 0) {
418                    pw.println();
419                    pw.print("Caused by: ");
420                }
421                pw.print(throwable.toString());
422            }
423            return sw.toString();
424        }

426        static void printStackTrace(Throwable throwable, PrintWriter s) {
427            Throwable[] stack = Util.toArray(throwable);
428            PrintWriter pw = new DummyPrintWriter(s);
429            for (int i = 0; i < stack.length; i++) {
430                if (i > 0) {
431                    pw.println("caused by");
432                }
433                stack[i].printStackTrace(pw);
434            }
435            pw.flush();
436        }

450        static void generateCommentBlock(
451                PrintWriter pw,
452                String name,
453                String text,
454                String comment)
455        {
456            final String indent = "    ";
457            pw.println(indent + "/**");
458            if (comment != null) {
459                fillText(pw, comment, indent + " * ", "", 70);
460                pw.println();
461                pw.println(indent + " *");
462            }
463            pw.print(indent + " * ");
464            fillText(pw,
465                    "<code>" + name + "</code> is '" + text + "'",
466                    indent + " * ", "", -1);
467            pw.println();
468            pw.println(indent + " */");
469        }

471        /**
472         * Returns the class name without its package name but with a locale
473         * extension, if applicable.
474         * For example, if class name is <code>happy.BirthdayResource</code>,
475         * and locale is <code>en_US</code>,
476         * returns <code>BirthdayResource_en_US</code>.
477         */
478        static String getClassNameSansPackage(String className, Locale locale) {
479            String s = className;
480            int lastDot = className.lastIndexOf('.');
481            if (lastDot >= 0) {
482                s = s.substring(lastDot + 1);
483            }
484            if (locale != null) {
485                s += '_' + locale.toString();
486            }
487            return s;
488        }

490        protected static String removePackage(String s)
491        {
492            int lastDot = s.lastIndexOf('.');
493            if (lastDot >= 0) {
494                s = s.substring(lastDot + 1);
495            }
496            return s;
497        }
