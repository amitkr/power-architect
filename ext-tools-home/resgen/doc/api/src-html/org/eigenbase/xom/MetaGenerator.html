061        /**
062         * This helper class contains all necessary information about a type.
063         * The information is collected here to help support inheritence and
064         * other advanced features.
065         */
066        private class TypeInfo
067        {
068            // XML definition of the type.  Includes defined attributes,
069            // content, and other information.
070            public MetaDef.Definition def;
071    
072            // Documentation and code, here for easy reference.
073            public String doc;
074            public String code;
075    
076            // Name of the class and associated XML tag.
077            public String name;
078            public String className;
079            public String tagName;
080    
081            // This array holds all attributes, inherited or otherwise, that may
082            // be used by this type.
083            public MetaDef.Attribute[] allAttributes;
084    
085            // This array holds all attributes that are overridden by this type.
086            public MetaDef.Attribute[] ovrAttributes;
087    
088            // This array holds all new attributes defined only in this type
089            // and not overriding any inherited attributes.
090            public MetaDef.Attribute[] newAttributes;
091    
092            // This array holds all content, inherited or otherwise, that may
093            // be used by this type.
094            public MetaDef.Content[] allContent;
095    
096            // This array holds all new content defined only in this type.
097            public MetaDef.Content[] newContent;
098    
099            // True if content is <Any> (either defined or inherited).
100            public boolean isAny;
101    
102            // True if content is <CData> (either defined or inherited).
103            public boolean isCData;
104    
105            // Reference to superclass info (if any)
106            public TypeInfo superInfo;
107    
108            // Class to use when importing elements.
109            public Class impClass;
110            public String impName; // e.g. "foo.MetaDef.Tag"

114            public TypeInfo(MetaDef.Definition elt)
115                throws XOMException
116            {
117                def = elt;
118    
119                // Get the name and superclass name
120                name = null;
121                String superName = null;
122                MetaDef.Attribute[] attributes = null;
123                MetaDef.Content[] content = null;
124                contentModel = "sequential";
125                if(elt instanceof MetaDef.Element) {
126                    MetaDef.Element element = (MetaDef.Element)elt;
127                    name = element.type;
128                    if(element.dtdName != null)
129                        tagName = element.dtdName;
130                    else
131                        tagName = prefix + name;
132                    superName = element._class;
133                    attributes = element.attributes;
134                    content = element.content;
135                    contentModel = element.contentModel;
136                    doc = element.doc;
137                    code = element.code;
138                    impClass = null;
139                    impName = null;
140                }
141                else if(elt instanceof MetaDef.Plugin) {
142                    name = ((MetaDef.Plugin)elt).type;
143                    tagName = prefix + name;
144                    superName = ((MetaDef.Plugin)elt)._class;
145                    attributes = ((MetaDef.Plugin)elt).attributes;
146                    content = new MetaDef.Content[0];
147                    doc = ((MetaDef.Plugin)elt).doc;
148                    code = ((MetaDef.Plugin)elt).code;
149                    impClass = null;
150                    impName = null;
151                }
152                else if(elt instanceof MetaDef.Class) {
153                    name = ((MetaDef.Class)elt)._class;
154                    tagName = "(%" + name + ";)";
155                    superName = ((MetaDef.Class)elt).superclass;
156                    attributes = ((MetaDef.Class)elt).attributes;
157                    content = ((MetaDef.Class)elt).content;
158                    doc = ((MetaDef.Class)elt).doc;
159                    code = ((MetaDef.Class)elt).code;
160                    impClass = null;
161                    impName = null;
162                }
163                else if(elt instanceof MetaDef.StringElement) {
164                    name = ((MetaDef.StringElement)elt).type;
165                    tagName = prefix + name;
166                    superName = null;
167                    attributes = new MetaDef.Attribute[0];
168                    content = new MetaDef.Content[0];
169                    doc = ((MetaDef.StringElement)elt).doc;
170                    code = null;
171                    impClass = null;
172                    impName = null;
173                }
174                else if (elt instanceof MetaDef.Import) {
175                    MetaDef.Import imp = (MetaDef.Import)elt;
176                    name = imp.type;
177                    if(imp.dtdName != null)
178                        tagName = imp.dtdName;
179                    else
180                        tagName = prefix + name;
181                    superName = null;
182                    attributes = new MetaDef.Attribute[0];
183                    content = new MetaDef.Content[0];
184                    doc = null;
185                    code = null;
186                    try {
187                        impName = imp.defPackage + "." + imp.defClass + "." + name;
188                        impClass = Class.forName(imp.defPackage + "."
189                                                 + imp.defClass + "$"
190                                                 + name);
191                    } catch(ClassNotFoundException ex) {
192    //                      throw new XOMException(
193    //                          "Import " + name + " references Java Class "
194    //                          + imp.defPackage + "." + imp.defClass
195    //                          + "." + name + " that does not exist.");
196                    }
197                } else {
198                    throw new XOMException("Illegal element type "
199                                           + elt.getClass().getName());
200                }
201                className = XOMUtil.capitalize(name);
202    
203                // Get the TypeInfo record for the superclass.  If we don't find
204                // it, we'll have to create it by looking up its definition.
205                superInfo = null;
206                if(superName != null) {
207                    superInfo = (TypeInfo)(infoMap.get(superName));
208                    if(superInfo == null) {
209                        MetaDef.Definition superDef =
210                            (MetaDef.Definition)(infoMap.get(superName));
211                        if(superDef == null)
212                            throw new XOMException(
213                                "Parent class " + superName + " of element "
214                                + name + " was never defined.");
215                        superInfo = new TypeInfo(superDef);
216                    }
217                }
218    
219                // Check for special content (<Any> or <CData>).  If we find it,
220                // it must be the only content defined.
221                boolean newAny = false;
222                boolean newCData = false;
223                if(content.length == 1) {
224                    if(content[0] instanceof MetaDef.CData)
225                        newCData = true;
226                    else if(content[0] instanceof MetaDef.Any)
227                        newAny = true;
228                }
229    
230                // Make sure that <Any> or <CData> occurs only by itself.
231                if(!newAny && !newCData) {
232                    for(int i=0; i<content.length; i++) {
233                        if(content[i] instanceof MetaDef.CData ||
234                           content[i] instanceof MetaDef.Any)
235                            throw new XOMException(
236                                "Type " + name + " defines <Any> or <CData> "
237                                + "content as well as other content.");
238                    }
239                }
240    
241                // Do we have a superclass/supertype?
242                if(superInfo == null) {
243                    // No supertype, so consider this type by itself.
244                    allAttributes = attributes;
245                    ovrAttributes = new MetaDef.Attribute[0];
246                    newAttributes = allAttributes;
247    
248                    if(newAny || newCData) {
249                        isAny = newAny;
250                        isCData = newCData;
251                        allContent = new MetaDef.Content[0];
252                    }
253                    else {
254                        isAny = isCData = false;
255                        allContent = content;
256                    }
257                    newContent = allContent;
258                }
259                else {
260                    // Reconcile attributes.
261                    Hashtable attrHash = new Hashtable();
262                    Hashtable ovrHash = new Hashtable();
263                    Vector allAttrs = new Vector();
264                    Vector ovrAttrs = new Vector();
265                    Vector newAttrs = new Vector();
266    
267                    for(int i=0; i<superInfo.allAttributes.length; i++) {
268                        attrHash.put(superInfo.allAttributes[i].name,
269                                     superInfo.allAttributes[i]);
270                    }
271                    for(int i=0; i<attributes.length; i++) {
272                        // Does the attribute already exist?
273                        MetaDef.Attribute inhAttr =
274                            (MetaDef.Attribute)(attrHash.get(attributes[i].name));
275                        if(inhAttr == null) {
276                            // attribute doesn't exist, so add to all and new.
277                            allAttrs.addElement(attributes[i]);
278                            newAttrs.addElement(attributes[i]);
279                        }
280                        else {
281                            // attribute does exist.  Type must match exactly.
282                            if(!(attributes[i].type.equals(inhAttr.type)))
283                                throw new XOMException(
284                                    "Element " + name + " inherits attribute "
285                                    + inhAttr.name + " of type " + inhAttr.type
286                                    + " but redefines it to be of type "
287                                    + attributes[i].type);
288    
289                            // Add to overridden vector and overridden hashtable
290                            ovrAttrs.addElement(attributes[i]);
291                            ovrHash.put(attributes[i].name,
292                                        attributes[i]);
293                        }
294                    }
295    
296                    // Add all non-overridden attributes to the allAttributes vector
297                    for(int i=0; i<superInfo.allAttributes.length; i++) {
298                        if(ovrHash.get(superInfo.allAttributes[i].name) == null)
299                            allAttrs.addElement(superInfo.allAttributes[i]);
300                    }
301    
302                    // Add all overridden attributes to the allAttributes vector
303                    for(int i=0; i<ovrAttrs.size(); i++)
304                        allAttrs.addElement(ovrAttrs.elementAt(i));
305    
306                    allAttributes = new MetaDef.Attribute[allAttrs.size()];
307                    for(int i=0; i<allAttributes.length; i++)
308                        allAttributes[i] =
309                            (MetaDef.Attribute)(allAttrs.elementAt(i));
310                    ovrAttributes = new MetaDef.Attribute[ovrAttrs.size()];
311                    for(int i=0; i<ovrAttributes.length; i++)
312                        ovrAttributes[i] =
313                            (MetaDef.Attribute)(ovrAttrs.elementAt(i));
314                    newAttributes = new MetaDef.Attribute[newAttrs.size()];
315                    for(int i=0; i<newAttributes.length; i++)
316                        newAttributes[i] =
317                            (MetaDef.Attribute)(newAttrs.elementAt(i));
318    
319                    // Reconcile content.  First check for specials.
320                    if(newAny || newCData) {
321                        if(superInfo.isAny || superInfo.isCData)
322                            throw new XOMException(
323                                "Element " + name + " both defines and inherits "
324                                + "<CData> or <Any> content.");
325                        if(superInfo.allContent.length > 0)
326                            throw new XOMException(
327                                "Element " + name + " inherits standard content "
328                                + "but defines <CData> or <Any> content.");
329                        isAny = newAny;
330                        isCData = newCData;
331                        allContent = new MetaDef.Content[0];
332                        newContent = new MetaDef.Content[0];
333                    }
334                    else if (superInfo.isAny || superInfo.isCData) {
335                        if(content.length > 0)
336                            throw new XOMException(
337                                "Element " + name + " inherits <CData> or <Any> "
338                                + "content but defines standard content.");
339                        isAny = superInfo.isAny;
340                        isCData = superInfo.isCData;
341                        allContent = new MetaDef.Content[0];
342                        newContent = new MetaDef.Content[0];
343                    }
344                    else {
345                        isAny = isCData = false;
346    
347                        // Overriding of content is forbidden.
348                        Hashtable contentHash = new Hashtable();
349                        Vector allContentVec = new Vector();
350                        for(int i=0; i<superInfo.allContent.length; i++) {
351                            contentHash.put(getContentName(superInfo.allContent[i]),
352                                            superInfo.allContent[i]);
353                            allContentVec.addElement(superInfo.allContent[i]);
354                        }
355                        for(int i=0; i<content.length; i++) {
356                            MetaDef.Content inhContent =
357                                (MetaDef.Content)
358                                (contentHash.get(getContentName(content[i])));
359                            if(inhContent != null)
360                                throw new XOMException(
361                                    "Content named " + getContentName(content[i])
362                                    + " defined in element " + name + " was "
363                                    + "already defined in an inherited element.");
364                            allContentVec.addElement(content[i]);
365                        }
366                        allContent = new MetaDef.Content[allContentVec.size()];
367                        for(int i=0; i<allContent.length; i++)
368                            allContent[i] =
369                                (MetaDef.Content)(allContentVec.elementAt(i));
370                        newContent = content;
371                    }
372                }
373    
374                // Add ourself to the hashtable if we're not already there
375                if(infoMap.get(name) == null)
376                    infoMap.put(name, this);
377            }

379            public void writeJavaClass(PrintWriter out)
380                throws XOMException
381            {
382                // Documentation first
383                if(doc != null)
384                    writeJavaDoc(out, 1, doc);
385    
386                // Then create the inner class.
387                String abs = (def instanceof MetaDef.Class) ? "abstract " :
388                    "";
389                out.print("\tpublic static " + abs + "class " + className + " extends ");
390                if(superInfo == null)
391                    out.println("org.eigenbase.xom.ElementDef");
392                else
393                    out.println(superInfo.className);
394                if (isAny) {
395                    out.println("\t\timplements org.eigenbase.xom.Any");
396                }
397                out.println("\t{");
398    
399                // Default constructor
400                out.println("\t\tpublic " + className + "()");
401                out.println("\t\t{");
402                out.println("\t\t}");
403                out.println();
404    
405                // org.eigenbase.xom.DOMWrapper Constructor
406                out.println("\t\tpublic " + className
407                            + "(org.eigenbase.xom.DOMWrapper _def)");
408                out.println("\t\t\tthrows org.eigenbase.xom.XOMException");
409                out.println("\t\t{");
410    
411                // Body of constructor.  Special case for completely empty
412                // model (no content and no attributes) to avoid warnings
413                // about unused things.
414                boolean mixed = contentModel.equals("mixed");
415                if(allContent.length == 0 && allAttributes.length == 0 &&
416                   !isAny && !isCData &&
417                    !(def instanceof MetaDef.Plugin))
418                    ;  // constructor has no body
419                else {
420                    if (def instanceof MetaDef.Element &&
421                        ((MetaDef.Element) def).keepDef.booleanValue()) {
422                        out.println("\t\t\tthis._def = _def;");
423                    }
424    
425                    out.println("\t\t\ttry {");
426    
427                    // Plugins: read defPackage and defClass here.
428                    if(def instanceof MetaDef.Plugin) {
429                        out.println("\t\t\t\tdefPackage = "
430                                    + "org.eigenbase.xom.DOMElementParser."
431                                    + "requiredDefAttribute("
432                                    + "_def, \"defPackage\", \"org.eigenbase.xom\");");
433                        out.println("\t\t\t\tdefClass = org.eigenbase.xom.DOMElementParser."
434                                    + "requiredDefAttribute("
435                                    + "_def, \"defClass\", null);");
436    
437                        // Get the enclosure class we'll be using
438                        out.println("\t\t\t\tClass _pluginClass = "
439                                    + "org.eigenbase.xom.DOMElementParser.getPluginClass("
440                                    + "defPackage, defClass);");
441                    }
442    
443                    // Create the parser.  If using a Plugin, parse from a
444                    // different enclosure class.
445                    out.print("\t\t\t\torg.eigenbase.xom.DOMElementParser _parser "
446                              + "= new org.eigenbase.xom.DOMElementParser("
447                              + "_def, ");
448                    if(def instanceof MetaDef.Plugin)
449                        out.println("\"\", _pluginClass);");
450                    else {
451                        if(model.prefix == null)
452                            out.print("\"\", ");
453                        else
454                            out.print("\"" + model.prefix + "\", ");
455                        out.println(model.className + ".class);");
456                    }
457    
458                    // This line is emitted to avoid unused warnings.
459                    out.println("\t\t\t\t_parser = _parser;");
460    
461                    // Define a temp array if any Array elements are used
462                    if(hasContentType(allContent, MetaDef.Array.class)) {
463                        out.println("\t\t\t\torg.eigenbase.xom.NodeDef[] "
464                                    + "_tempArray = null;");
465                        out.println("\t\t\t\t_tempArray = _tempArray;");
466                    }
467    
468                    // Generate statements to read in all attributes.
469                    for(int i=0; i<allAttributes.length; i++)
470                        writeJavaGetAttribute(out, allAttributes[i]);
471    
472                    // Generate statements to read in all content.
473                    if(def instanceof MetaDef.Plugin)
474                        writeJavaGetPluginContent(out, mixed);
475                    else if(isAny)
476                        writeJavaGetAnyContent(out, mixed);
477                    else if(isCData)
478                        writeJavaGetCDataContent(out);
479                    else
480                        for(int i=0; i<allContent.length; i++)
481                            writeJavaGetContent(out, allContent[i]);
482    
483                    out.println("\t\t\t} catch(org.eigenbase.xom.XOMException _ex) {");
484                    out.println("\t\t\t\tthrow new org.eigenbase.xom.XOMException("
485                                + "\"In element '\" + getName() + \"': \" + _ex.getMessage());");
486                    out.println("\t\t\t}");
487                }
488    
489                // Finish the constructor
490                out.println("\t\t}");
491                out.println();
492    
493                // Declare all new attributes
494                for(int i=0; i<newAttributes.length; i++)
495                    writeJavaDeclareAttribute(out, newAttributes[i]);
496                if(def instanceof MetaDef.Plugin)
497                    writeJavaDeclarePluginAttributes(out);
498                if (def instanceof MetaDef.Element &&
499                    ((MetaDef.Element) def).keepDef.booleanValue()) {
500                    out.println("\t\tpublic org.eigenbase.xom.DOMWrapper _def;");
501                }
502                out.println();
503    
504                // Declare all new content
505                if(def instanceof MetaDef.Plugin)
506                    writeJavaDeclarePluginContent(out, mixed);
507                else if(isAny)
508                    writeJavaDeclareAnyContent(out, mixed);
509                else if(isCData)
510                    writeJavaDeclareCDataContent(out);
511                else
512                    for(int i=0; i<newContent.length; i++)
513                        writeJavaDeclareContent(out, newContent[i]);
514                out.println();
515    
516                // Create the getName() function
517                out.println("\t\tpublic String getName()");
518                out.println("\t\t{");
519                out.println("\t\t\treturn \"" + className + "\";");
520                out.println("\t\t}");
521                out.println();
522    
523                // Create the display() function
524                out.println("\t\tpublic void display(java.io.PrintWriter _out, "
525                            + "int _indent)");
526                out.println("\t\t{");
527                if(def instanceof MetaDef.Class && !isAny && !isCData &&
528                   allContent.length == 0 && allAttributes.length == 0)
529                    ;
530                else
531                    out.println("\t\t\t_out.println(getName());");
532                for(int i=0; i<allAttributes.length; i++)
533                    writeJavaDisplayAttribute(out, allAttributes[i]);
534                if(def instanceof MetaDef.Plugin)
535                    writeJavaDisplayPluginAttributes(out);
536    
537                if(def instanceof MetaDef.Plugin)
538                    writeJavaDisplayPluginContent(out);
539                else if(isAny)
540                    writeJavaDisplayAnyContent(out);
541                else if(isCData)
542                    writeJavaDisplayCDataContent(out);
543                else
544                    for(int i=0; i<allContent.length; i++)
545                        writeJavaDisplayContent(out, allContent[i]);
546                out.println("\t\t}");
547    
548                // Create the displayXML() function
549                out.println("\t\tpublic void displayXML("
550                            + "org.eigenbase.xom.XMLOutput _out, "
551                            + "int _indent)");
552                out.println("\t\t{");
553                out.println("\t\t\t_out.beginTag(\""
554                            + tagName + "\", "
555                            + "new org.eigenbase.xom.XMLAttrVector()");
556                for(int i=0; i<allAttributes.length; i++)
557                    writeJavaDisplayXMLAttribute(out, allAttributes[i]);
558                if(def instanceof MetaDef.Plugin)
559                    writeJavaDisplayXMLPluginAttributes(out);
560                out.println("\t\t\t\t);");
561    
562                if(def instanceof MetaDef.Plugin)
563                    writeJavaDisplayXMLPluginContent(out);
564                else if(isAny)
565                    writeJavaDisplayXMLAnyContent(out);
566                else if(isCData)
567                    writeJavaDisplayXMLCDataContent(out);
568                else
569                    for(int i=0; i<allContent.length; i++)
570                        writeJavaDisplayXMLContent(out, allContent[i]);
571    
572                out.println("\t\t\t_out.endTag(\"" + tagName + "\");");
573                out.println("\t\t}");
574    
575                // Create the displayDiff() function
576                out.println("\t\tpublic boolean displayDiff("
577                            + "org.eigenbase.xom.ElementDef _other, "
578                            + "java.io.PrintWriter _out, "
579                            + "int _indent)");
580                out.println("\t\t{");
581                out.println("\t\t\tboolean _diff = true;");
582                if(allAttributes.length > 0 ||
583                   allContent.length > 0 || isAny || isCData ||
584                   def instanceof MetaDef.Plugin)
585                    out.println("\t\t\t" + className + " _cother = ("
586                                + className + ")_other;");
587                for(int i=0; i<newAttributes.length; i++)
588                    writeJavaDisplayDiffAttribute(out, allAttributes[i]);
589                if(def instanceof MetaDef.Plugin)
590                    writeJavaDisplayDiffPluginAttributes(out);
591    
592                if(def instanceof MetaDef.Plugin)
593                    writeJavaDisplayDiffPluginContent(out);
594                else if(isAny)
595                    writeJavaDisplayDiffAnyContent(out);
596                else if(isCData)
597                    writeJavaDisplayDiffCDataContent(out);
598                else
599                    for(int i=0; i<allContent.length; i++)
600                        writeJavaDisplayDiffContent(out, allContent[i]);
601                out.println("\t\t\treturn _diff;");
602                out.println("\t\t}");
603    
604                // Add the code section, if defined
605                if(code != null)
606                    writeJavaCode(out, 2, code);
607    
608                // Complete the class definition and finish with a blank.
609                out.println("\t}");
610                out.println();
611            }
612        }

614        /**
615         * Get the name of any piece of content of any type.
616         * @return the name of the piece of content.
617         * @throws XOMException if the content is <Any> or <CData>.
618         */
619        private static String getContentName(MetaDef.Content content)
620            throws XOMException
621        {
622            if(content instanceof MetaDef.Object) {
623                return ((MetaDef.Object)content).name;
624            }
625            else if (content instanceof MetaDef.Array) {
626                return ((MetaDef.Array)content).name;
627            }
628            else {
629                throw new XOMException(
630                    "Content of type " + content.getClass().getName()
631                    + " does not have a name.");
632            }
633        }

635        /**
636         * Return the TypeInfo class associated with the given name.
637         *
638         * @post fail == false || return != null
639         * @exception XOMException if the type has not been defined
640         */
641        public TypeInfo getTypeInfo(String name, boolean fail)
642            throws XOMException
643        {
644            TypeInfo info = (TypeInfo) infoMap.get(name);
645            if (info == null && fail == true) {
646                throw new XOMException(
647                    "Type " + name + " does not exist.");
648            }
649            return info;
650        }

663        protected MetaGenerator(String xmlFile, boolean testMode, String className)
664                throws XOMException, IOException {
665            this.testMode = testMode;
666            // Create a non-validating XML parser to parse the file
667            FileInputStream in = new FileInputStream(xmlFile);
668            Parser parser = XOMUtil.createDefaultParser();
669            try {
670                DOMWrapper def = parser.parse(in);
671                model = new MetaDef.Model(def);
672            } catch (XOMException ex) {
673                throw new XOMException(ex, "Failed to parse XML file: " + xmlFile);
674            }
675    
676            // check that class names are consistent
677            if (className != null) {
678                if (model.className == null) {
679                    model.className = className;
680                } else {
681                    String modelClassName = model.className;
682                    if (model.packageName != null &&
683                        !model.packageName.equals("")) {
684                        modelClassName = model.packageName + "." +
685                            model.className;
686                    }
687                    if (!className.equals(modelClassName)) {
688                        throw new XOMException(
689                            "className parameter (" + className +
690                            ") is inconsistent with model's packageName and " +
691                            "className attributes (" + modelClassName + ")");
692                    }
693                }
694            }
695    
696            // Construct the meta model from its XML description
697            prefix = model.prefix;
698            if(prefix == null)
699                prefix = "";
700    
701            // Setup the Hashtable maps
702            initKeywordMap();
703            initTypeMap();
704            initSubclassMap();
705        }

769        /**
770         * All Elements in the meta model have an associated type name which
771         * identifies the element.  The type map allows the XMLDef.ElementType
772         * object describing an element to be retrieved from its name.  It is
773         * used to resolve references to element type names appearing
774         * throughout a model.
775         */
776        private void initTypeMap()
777            throws XOMException
778        {
779            typeMap = new Hashtable();
780            allTypes = new Vector();
781            for(int i=0; i<model.elements.length; i++) {
782                MetaDef.Definition elt = model.elements[i];
783                String name = null;
784                if(elt instanceof MetaDef.Element)
785                    name = ((MetaDef.Element)elt).type;
786                else if(elt instanceof MetaDef.Plugin)
787                    name = ((MetaDef.Plugin)elt).type;
788                else if(elt instanceof MetaDef.Class)
789                    name = ((MetaDef.Class)elt)._class;
790                else if(elt instanceof MetaDef.StringElement)
791                    name = ((MetaDef.StringElement)elt).type;
792                else if(elt instanceof MetaDef.Import)
793                    name = ((MetaDef.Import)elt).type;
794                else
795                    throw new XOMException("Illegal element type "
796                                              + elt.getClass().getName());
797                typeMap.put(name, elt);
798                allTypes.addElement(name);
799            }
800    
801            infoMap = new Hashtable();
802            for(int i=0; i<model.elements.length; i++) {
803                // Get the element
804                MetaDef.Definition elt = model.elements[i];
805    
806                // Construct the new TypeInfo object and add to the hashtable
807                TypeInfo info = new TypeInfo(elt);
808                infoMap.put(info.name, info);
809            }
810        }

812        /**
813         * In a few cases, a complete list of all subclasses of a class
814         * object is required.  The subclass map maps each class object
815         * (identified by its name) to a Vector containing all of its
816         * subclasses.  Currently, all subclasses must be Element types.
817         */
818        private void initSubclassMap()
819            throws XOMException
820        {
821            subclassMap = new Hashtable();
822    
823            // First, iterate through all Class elements in the model,
824            // initializing a location in the hashtable for each.
825            for(int i=0; i<model.elements.length; i++) {
826                MetaDef.Definition elt = model.elements[i];
827                if(elt instanceof MetaDef.Class) {
828                    MetaDef.Class _class = (MetaDef.Class)elt;
829                    subclassMap.put(_class._class, new Vector());
830                }
831            }
832    
833            // Now, iterate through all Element elements in the model.
834            // For each one, go through all of its superclasses and add itself to
835            // the vector of each.
836            // If a class is not found, it is an error.
837            for(int i=0; i<model.elements.length; i++) {
838                MetaDef.Definition elt = model.elements[i];
839                if(elt instanceof MetaDef.Element) {
840                    MetaDef.Element elem = (MetaDef.Element)elt;
841                    TypeInfo info = getTypeInfo(elem.type, true);
842                    addToSubclassMap(elem, info);
843                }
844            }
845        }

847        /**
848         * Helper method for initSubclassMap:
849         * Add this element to the subclass map for each superclass of info.
850         */
851        private void addToSubclassMap(MetaDef.Element elem, TypeInfo info)
852            throws XOMException
853        {
854            while(info.superInfo != null) {
855                // Add the element to this class's vector.
856                Vector vec = (Vector)(subclassMap.get(info.superInfo.name));
857                if(vec == null)
858                    throw new XOMException("Class " + info.superInfo.name +
859                                              " of element " + elem.type
860                                              + " is not defined.");
861                vec.addElement(elem);
862    
863                // Add to all superclasses as well
864                info = info.superInfo;
865            }
866        }

868        /**
869         * Create all files associated with the metamodel, including a Java class
870         * and a DTD file.  The DTD is primarily for reference--it will not work
871         * if any advanced features (plugins, includes) are used.
872         * @param outputDirName the output directory in which to generate the files.
873         */
874        public void writeFiles(String outputDirName, String dtdFileName)
875            throws XOMException, IOException
876        {
877            // Compute the output file names
878            if (dtdFileName != null) {
879                if (model.dtdName == null) {
880                    model.dtdName = dtdFileName;
881                } else {
882                    if (!dtdFileName.equals(model.dtdName)) {
883                        throw new XOMException(
884                            "dtdFileName parameter (" + dtdFileName +
885                            ") is inconsistent with model's dtdName " +
886                            "attribute (" + model.dtdName + ")");
887                    }
888                }
889            }
890            File javaOutputDir = new File(outputDirName);
891    
892            if (!testMode &&
893                model.packageName != null &&
894                !model.packageName.equals("")) {
895                javaOutputDir = new File(
896                        javaOutputDir, model.packageName.replace('.',fileSep));
897            }
898            File javaFile = new File(javaOutputDir, model.className + ".java");
899            File outputDir = javaFile.getParentFile();
900            File dtdFile = new File(outputDir, model.dtdName);
901    
902            // If the output file is MetaDef.java, and we start writing to
903            // MetaDef.java before we have loaded MetaDef.class, the system thinks
904            // that the class is out of date.  So load MetaDef.class before that
905            // point.
906            XOMUtil.discard(new MetaDef());
907    
908            // Create directories if necessary.
909            outputDir.mkdir();
910    
911            // Open the files for writing
912            FileWriter dtdWriter = new FileWriter(dtdFile);
913            PrintWriter dtdOut = new PrintWriter(dtdWriter);
914            FileWriter javaWriter = new FileWriter(javaFile);
915            PrintWriter javaOut = new PrintWriter(javaWriter);
916    
917            if(!testMode)
918                System.out.println("Writing " + dtdFile);
919            writeDtd(dtdOut);
920            dtdOut.flush();
921            dtdWriter.close();
922    
923            if(!testMode)
924                System.out.println("Writing " + javaFile);
925            writeJava(javaOut);
926            javaOut.flush();
927            javaWriter.close();
928    
929            if(!testMode)
930                System.out.println("Done");
931        }

933        public void writeDtd(PrintWriter out)
934            throws XOMException
935        {
936            // Write header information for the dtd
937            out.println("<!--");
938            out.println("     This dtd file was automatically generated from "
939                      + "XOM model " + model.name + ".");
940            out.println("     Do not edit this file by hand.");
941            out.println("  -->");
942            out.println();
943    
944            // Write toplevel documentation here
945            writeDtdDoc(out, model.doc);
946    
947            // For each CLASS definition, write an entity definition.  These must
948            // be done before regular elements because entities must be defined
949            // before use.
950            for(int i=0; i<model.elements.length; i++) {
951                if(model.elements[i] instanceof MetaDef.Class) {
952                    writeDtdEntity(out, (MetaDef.Class)(model.elements[i]));
953                }
954            }
955    
956            // Write each element in turn
957            for(int i=0; i<model.elements.length; i++)
958                writeDtdElement(out, model.elements[i]);
959        }

961        public void writeJava(PrintWriter out)
962            throws XOMException
963        {
964            // Write header information for the java file
965            out.println("/" + "*");
966            out.println("/" + "/ This java file was automatically generated");
967            out.println("/" + "/ from XOM model '" + model.name + "'");
968            if(!testMode)
969                out.println("/" + "/ on " + new Date().toString());
970            out.println("/" + "/ Do not edit this file by hand.");
971            out.println("*" + "/");
972            out.println();
973    
974            if (!testMode &&
975                !(model.packageName == null || model.packageName.equals(""))) {
976                out.println("package " + model.packageName + ";");
977            }
978    
979            // Write the toplevel documentation for the package.  This becomes
980            // the toplevel documentation for the class and is also placed at
981            // the top of the Dtd.
982            String extraDoc = newLine + "<p>This class was generated from XOM model '"
983                + model.name + "' on " + new Date().toString();
984            if(testMode)
985                extraDoc = "";
986    
987            writeJavaDoc(out, 0, model.doc + extraDoc);
988    
989            // Begin the class.  Include a getXMLDefClass() function which
990            // simply returns this class.
991            out.println("public class " + model.className + " {");
992            out.println();
993            out.println("\tpublic static java.lang.Class getXMLDefClass()");
994            out.println("\t{");
995            out.println("\t\treturn " + model.className + ".class;");
996            out.println("\t}");
997            out.println();
998    
999            // Create a static member that names all Elements that may be
1000            // used within this class.
1001            out.println("\tpublic static String[] _elements = {");
1002            for(int i=0; i<allTypes.size(); i++) {
1003                String type = (String) allTypes.elementAt(i);
1004                out.print("\t\t\"" + type + "\"");
1005                if(i<allTypes.size()-1)
1006                    out.println(",");
1007                else
1008                    out.println();
1009            }
1010            out.println("\t};");
1011            out.println();
1012    
1013            // Create an inner class for each Class/Object definition.
1014            for(int i=0; i<model.elements.length; i++)
1015                writeJavaElement(out, model.elements[i]);
1016    
1017            // End the class
1018            out.println();
1019            out.println("}");
1020        }

1022        /**
1023         * Writes an entity definition based on a defined Class.  Because entity
1024         * definitions must appear before use in a DTD, this function must be
1025         * called for each defined class before processing the rest of the model.
1026         * @param out PrintWriter to write the DTD.
1027         * @param _class Class definition on which the Entity will be based.
1028         */
1029        private void writeDtdEntity(PrintWriter out, MetaDef.Class _class)
1030        {
1031            // Documentation first
1032            if(_class.doc != null)
1033                writeDtdDoc(out, _class.doc);
1034    
1035            // Lookup the subclass vector for this class.  Use this to generate
1036            // the entity definition.
1037            Vector subclassVec = (Vector)(subclassMap.get(_class._class));
1038            out.print("<!ENTITY % " + _class._class + " \"");
1039            if(subclassVec == null)
1040                throw new AssertFailure(
1041                    "Missing subclass vector for class " + _class._class);
1042            for(int i=0; i<subclassVec.size(); i++) {
1043                MetaDef.Element elem =
1044                    (MetaDef.Element)(subclassVec.elementAt(i));
1045    
1046                // Print the dtd version of the element name
1047                if(elem.dtdName != null)
1048                    out.print(elem.dtdName);
1049                else
1050                    out.print(prefix + elem.type);
1051    
1052                if(i<subclassVec.size()-1)
1053                    out.print("|");
1054            }
1055            out.println("\">");
1056            out.println();
1057        }

1059        private void writeDtdElement(PrintWriter out, MetaDef.Definition elt)
1060            throws XOMException
1061        {
1062            // What is written into the dtd depends on the class of elt.
1063            if(elt instanceof MetaDef.Element) {
1064                // Get the info class for this element.
1065                MetaDef.Element element = (MetaDef.Element)elt;
1066                TypeInfo info = getTypeInfo(element.type, false);
1067                if(info == null)
1068                    throw new AssertFailure("Element type " + element.type + " is missing from the "
1069                                     + "type map.");
1070    
1071                // Documentation first
1072                if(element.doc != null)
1073                    writeDtdDoc(out, element.doc);
1074    
1075                // Then content model.  Special case empty models.
1076                out.print("<!ELEMENT " + info.tagName + " ");
1077                if(info.allContent.length == 0 && !info.isAny && !info.isCData)
1078                    out.print("EMPTY");
1079                else {
1080                    if(info.isAny)
1081                        out.print("ANY");
1082                    else if(info.isCData)
1083                        out.print("(#PCDATA)");
1084                    else {
1085                        out.print("(");
1086                        for(int i=0; i<info.allContent.length; i++) {
1087                            writeDtdContent(out, info.allContent[i]);
1088                            if(i<info.allContent.length-1)
1089                                out.print(",");
1090                        }
1091                        out.print(")");
1092                    }
1093                }
1094                out.println(">");
1095    
1096                // Finally, attribute list
1097                if(info.allAttributes.length > 0) {
1098                    out.println("<!ATTLIST " + info.tagName);
1099                    for(int i=0; i<info.allAttributes.length; i++)
1100                        writeDtdAttribute(out, info.allAttributes[i]);
1101                    out.println(">");
1102                }
1103    
1104                // Finish with a blank
1105                out.println();
1106            }
1107            else if (elt instanceof MetaDef.Class) {
1108                // Do nothing--entities are handled ahead of time.
1109            }
1110            else if (elt instanceof MetaDef.StringElement) {
1111                // Get the info class for this element.
1112                MetaDef.StringElement element = (MetaDef.StringElement)elt;
1113                TypeInfo info = (TypeInfo)(infoMap.get(element.type));
1114                if(info == null)
1115                    throw new AssertFailure("StringElement type " + element.type +
1116                                     " is missing from the "
1117                                     + "type map.");
1118    
1119                // Documentation first
1120                if(element.doc != null)
1121                    writeDtdDoc(out, element.doc);
1122    
1123                // Then content model.  It is always (#PCDATA).
1124                out.println("<!ELEMENT " + info.tagName + " (#PCDATA)>");
1125                out.println();
1126            }
1127            else if (elt instanceof MetaDef.Plugin) {
1128                // Get the info class for this element.
1129                MetaDef.Plugin plugin = (MetaDef.Plugin)elt;
1130                TypeInfo info = (TypeInfo)(infoMap.get(plugin.type));
1131                if(info == null)
1132                    throw new AssertFailure("Plugin element " + plugin.type +
1133                                     " is missing from the "
1134                                     + "type map.");
1135    
1136                // Documentation first
1137                if(plugin.doc != null)
1138                    writeDtdDoc(out, plugin.doc);
1139    
1140                // Then content model.  It is always ANY.
1141                out.println("<!ELEMENT " + info.tagName + " ANY>");
1142    
1143                // Finally, attribute list.  Don't allow use of plugin reserved
1144                // attributes defPackage and defClass.
1145                out.println("<!ATTLIST " + info.tagName);
1146                for(int i=0; i<info.allAttributes.length; i++) {
1147                    if(info.allAttributes[i].name.equals("defPackage") ||
1148                       info.allAttributes[i].name.equals("defClass"))
1149                        throw new XOMException(
1150                            "The attribute \"" + info.allAttributes[i].name
1151                            + "\" is reserved and may not be redefined in "
1152                            + "or inherited by a Plugin.");
1153                    writeDtdAttribute(out, info.allAttributes[i]);
1154                }
1155    
1156                // Add attribute definitions for defPackage and defClass
1157                out.println("defPackage CDATA \"org.eigenbase.xom\"");
1158                out.println("defClass CDATA #REQUIRED");
1159    
1160                // Complete the attribute list
1161                out.println(">");
1162                out.println();
1163            }
1164            else if (elt instanceof MetaDef.Import) {
1165                // Get the info class for this element.
1166                MetaDef.Import imp = (MetaDef.Import)elt;
1167                TypeInfo info = getTypeInfo(imp.type, true);
1168    
1169                // Imports can't really be handled, so just generate a placeholder
1170                // ANY element for show.
1171                out.println("<!ELEMENT " + info.name + " ANY>");
1172                out.println();
1173            }
1174            else {
1175                throw new XOMException("Unrecognized element type definition: "
1176                                          + elt.getClass().getName());
1177            }
1178        }

1287        private void writeDtdContent(PrintWriter out, MetaDef.Content content)
1288            throws XOMException
1289        {
1290            if(content instanceof MetaDef.Object) {
1291                MetaDef.Object obj = (MetaDef.Object)content;
1292                TypeInfo info = (TypeInfo)(infoMap.get(obj.type));
1293                if(info == null)
1294                    throw new XOMException(
1295                        "Object " + obj.name + " has undefined type "
1296                        + obj.type);
1297                out.print(info.tagName);
1298                if(!obj.required.booleanValue())

1958        /**
1959         * Main function for MetaGenerator. Arguments:
1960         * <ol>
1961         * <li>Name of XML file describing input model.
1962         * <li>Name of output file directory.
1963         * </ol>
1964         */
1965        public static void main(String[] args)
1966        {
1967            int firstArg = 0;
1968            boolean testMode = false;
1969            if (firstArg < args.length && args[firstArg].equals("-debug")) {
1970                System.err.println("MetaGenerator pausing for debugging.  "
1971                                   + "Attach your debugger "
1972                                   + "and press return.");
1973                try {
1974                    System.in.read();
1975                    firstArg++;
1976                }
1977                catch(IOException ex) {
1978                    // Do nothing
1979                }
1980            }
1981            if (firstArg < args.length && args[firstArg].equals("-test")) {
1982                System.err.println("Ignoring package name.");
1983                testMode = true;
1984                firstArg++;
1985            }
1986    
1987            if(args.length != 2 + firstArg) {
1988                System.err.println(
1989                    "Usage: java MetaGenerator [-debug] [-test] " +
1990                    "<XML model file> <output directory>");
1991                System.exit(2);
1992            }
1993    
1994            try {
1995                MetaGenerator generator = new MetaGenerator(
1996                    args[0+firstArg], testMode);
1997                generator.writeFiles(args[1+firstArg], null);
1998                generator.writeOutputs();
1999            }
2000            catch(XOMException ex) {
2001                System.err.println("Generation of model failed:");
2002                System.err.println(ex.toString());
2003                ex.printStackTrace();
2004                System.exit(1);
2005            }
2006            catch(IOException ex) {
2007                System.err.println("Generation of model failed:");
2008                System.err.println(ex.toString());
2009                ex.printStackTrace();
2010                System.exit(1);
2011            }
2012        }
