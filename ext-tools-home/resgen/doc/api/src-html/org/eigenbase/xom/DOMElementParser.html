033    /**
034     * DOMElementParser is a utility wrapper around DOMWrapper.
035     * implements a parseable stream of child DOMWrappers and also provides
036     * validation on an XML document beyond the DTD.
037     */
038    public class DOMElementParser {
039    
040        private DOMWrapper wrapper;
041        private DOMWrapper[] children;
042        private int currentIndex;
043        private DOMWrapper currentChild;
044    
045        private int optionIndex;
046        private String prefix;
047        private Class enclosure;
048    
049        /**
050         * Constructs an new ElementParser based on an Element of the XML parse
051         * tree wrapped in a DOMWrapper, and a prefix (to be applied to all element
052         * tags except the root), and the name of the enclosing class.
053         * @param wrapper a DOMWrapper representing the section of the XML parse tree
054         * to traverse.
055         */
056        public DOMElementParser(DOMWrapper wrapper, String prefix, Class enclosure)
057            throws XOMException
058        {
059            this.wrapper = wrapper;
060            children = wrapper.getElementChildren();
061            currentIndex = 0;
062            currentChild = null;
063            getNextElement();
064    
065            this.prefix = prefix;
066            if (prefix == null) {
067                this.prefix = "";
068            }
069            this.enclosure = enclosure;
070        }
071    
072        /**
073         * Private helper function to retrieve the next child element in sequence.
074         * @return the next element, or null if the enumerator has no more
075         * elements to return.
076         */
077        private void getNextElement()
078        {
079            if (currentIndex >= children.length) {
080                currentChild = null;
081            } else {
082                currentChild = children[currentIndex++];
083            }
084        }
085    
086        /**
087         * Private helper function to verify that the next element matches a
088         * specific name.
089         * @param name name of the element to match.  Names are not case-sensitive.
090         * @throws XOMException if there is no current element or the names do
091         * not match.
092         */
093        private void requiredName(String name)
094            throws XOMException
095        {
096            String augName = prefix + name;
097            if (currentChild == null) {
098                throw new XOMException("Expected <"+ augName + "> but found "
099                                          + "nothing.");
100            } else if(!augName.equalsIgnoreCase(currentChild.getTagName())) {
101                throw new XOMException("Expected <" + augName + "> but found <"
102                                          + currentChild.getTagName() + ">");
103            }
104        }
105    
106        /**
107         * Private helper function to determine if the next element has the
108         * specified name.
109         * @return true if the next element's name matches <i>name</i>.  Matching
110         * is not case-sensitive.  Returns false if there is no next element or
111         * if the names don't match.
112         */
113        private boolean optionalName(String name)
114        {
115            String augName = prefix + name;
116            if (currentChild == null) {
117                return false;
118            } else if (augName.equalsIgnoreCase(currentChild.getTagName())) {
119                return true;
120            } else {
121                return false;
122            }
123        }

125        /**
126         * Returns the enclosure class associated with clazz, or falls back on
127         * the fixed enclosure if none can be found.
128         */
129        private Class getEnclosureClass(Class clazz)
130        {
131            // Instead of using a fixed enclosure, derive it from the given Class.
132            // If we can't figure it out, just use the given enclosure instead.
133            Class thisEnclosure = enclosure;
134            String className = clazz.getName();
135            int dollarPos = className.indexOf('$');
136            if (dollarPos >= 0) {
137                String encName = className.substring(0, dollarPos);
138                try {
139                    thisEnclosure = Class.forName(encName);
140                } catch (ClassNotFoundException ex) {
141                    throw new AssertFailure("Enclosure class " + encName
142                                     + " not found.");
143                }
144            }
145            return thisEnclosure;
146        }

148        /**
149         * Private helper function to determine if the next element's corresponding
150         * definition class is a subclass of the given class.  This may be used
151         * to detect if a name matches a class.
152         * @param clazz the class to match the next element against.
153         * @return true if the next element's name matches the given class, false
154         * otherwise.
155         * @throws XOMException if the next name is invalid (either doesn't
156         * start with DM or has no associated definition class).
157         */
158        private boolean nameMatchesClass(Class clazz)
159            throws XOMException
160        {
161            // Get the next name.  It must start with the set prefix, and it must
162            // match a definition in the enclosure class.
163            Class thisEnclosure = getEnclosureClass(clazz);
164            Class nextClass = ElementDef.getElementClass(currentChild,
165                                                         thisEnclosure,
166                                                         prefix);
167    
168            // Determine if nextClass is a subclass of clazz.  Return true if so.
169            return nextClass != null &&
170                    clazz.isAssignableFrom(nextClass);
171        }

253        /**
254         * This private helper function formats a list of element names into
255         * a readable string for error messages.
256         */
257        private String formatOption(String[] elementNames)
258        {
259            StringBuffer sbuf = new StringBuffer();
260            for (int i = 0; i < elementNames.length; i++) {
261                sbuf.append("<DM" + prefix);
262                sbuf.append(elementNames[i]);
263                sbuf.append(">");
264                if (i < elementNames.length-1) {
265                    sbuf.append(" or ");
266                }
267            }
268            return sbuf.toString();
269        }

271        /**
272         * This function retrieves a required element which may have one of a
273         * number of names.  The parser is advanced after the read.
274         * @param elementNames an array of allowed names.  Names are compared in
275         * a case-insensitive fashion.
276         * @return the first element with one of the given names.
277         * @throws XOMException if there are no more elements to read or if
278         * the next element's name is not in the elementNames list.
279         */
280        public DOMWrapper requiredOption(String[] elementNames)
281            throws XOMException
282        {
283            if (currentChild == null) {
284                throw new XOMException("Expecting "
285                                          + formatOption(elementNames)
286                                          + " but found nothing.");
287            } else {
288                for (int i = 0; i < elementNames.length; i++) {
289                    String augName = "DM" + elementNames[i];
290                    if(augName.equalsIgnoreCase(
291                        currentChild.getTagName().toString())) {
292                        DOMWrapper prevWrapper = currentChild;
293                        getNextElement();
294                        optionIndex = i;
295                        return prevWrapper;
296                    }
297                }
298    
299                // If we got here, no names match.
300                throw new XOMException("Expecting "
301                                          + formatOption(elementNames)
302                                          + " but found <"
303                                          + currentChild.getTagName()
304                                          + ">.");
305            }
306        }

308        /**
309         * This function retrieves a required Element of a specific class
310         * from this parser, advancing the parser after the read.
311         * The class must be derived from ElementDef.
312         */
313        public NodeDef requiredClass(Class classTemplate)
314            throws XOMException
315        {
316            // The name must match the class.
317            if (!nameMatchesClass(classTemplate)) {
318                throw new XOMException("element <" + currentChild.getTagName()
319                                          + "> does not match expected class "
320                                          + classTemplate.getName());
321            }
322    
323            // Get the class corresponding to the current tag
324            Class currentClass = ElementDef.getElementClass(currentChild,
325                                                            enclosure, prefix);
326    
327            // Get the element
328            DOMWrapper prevWrapper = currentChild;
329            getNextElement();
330    
331            // Construct an ElementDef of the correct class from the element
332            return ElementDef.constructElement(prevWrapper, currentClass);
333        }

520        /**
521         * This function retrieves a collection of elements with the given name,
522         * returning them as an array.
523         * @param elemName the element name.
524         * @param min the minimum number of elements required in the array.  Set
525         * this parameter to 0 to indicate no minimum.
526         * @param max the maximum number of elements allowed in the array.  Set
527         * this parameter to 0 to indicate no maximum.
528         * @return an Element array containing the discovered elements.
529         * @throws XOMException if there are fewer than min or more than max
530         * elements with the name <i>elemName</i>.
531         */
532        public DOMWrapper[] optionalArray(String elemName, int min, int max)
533            throws XOMException
534        {
535            // First, read the appropriate elements into a vector.
536            Vector vec = new Vector();
537            String augName = "DM" + elemName;
538            while(currentChild != null &&
539                  augName.equalsIgnoreCase(currentChild.getTagName())) {
540                vec.addElement(currentChild);
541                getNextElement();
542            }
543    
544            // Now, check for size violations
545            if (min > 0 && vec.size() < min) {
546                throw new XOMException("Expecting at least " + min + " <"
547                                          + elemName + "> but found " + vec.size());
548            }
549            if (max > 0 && vec.size() > max) {
550                throw new XOMException("Expecting at most " + max + " <"
551                                          + elemName + "> but found " +
552                                          vec.size());
553            }
554    
555            // Finally, convert to an array and return.
556            DOMWrapper[] retval = new DOMWrapper[vec.size()];
557            for (int i = 0; i < retval.length; i++)
558                retval[i] = (DOMWrapper)(vec.elementAt(i));
559            return retval;
560        }

562        /**
563         * This function retrieves a collection of elements which are subclasses of
564         * the given class, returning them as an array.  The array will contain
565         * ElementDef objects automatically constructed to be of the correct class.
566         * @param elemClass the element class.
567         * @param min the minimum number of elements required in the array.  Set
568         * this parameter to 0 to indicate no minimum.
569         * @param max the maximum number of elements allowed in the array.  Set
570         * this parameter to 0 to indicate no maximum.
571         * @return an ElementDef array containing the discovered elements.
572         * @throws XOMException if there are fewer than min or more than max
573         * elements with the name <i>elemName</i>.
574         */
575        public NodeDef[] classArray(Class elemClass, int min, int max)
576            throws XOMException
577        {
578            // Instead of using a fixed enclosure, derive it from the given Class.
579            // If we can't figure it out, just use the given enclosure instead.
580            Class thisEnclosure = getEnclosureClass(elemClass);
581    
582            // First, read the appropriate elements into a vector.
583            Vector vec = new Vector();
584            while(currentChild != null &&
585                  nameMatchesClass(elemClass)) {
586                vec.addElement(currentChild);
587                getNextElement();
588            }
589    
590            // Now, check for size violations
591            if (min > 0 && vec.size() < min) {
592                throw new XOMException("Expecting at least " + min + " <"
593                                          + elemClass.getName()
594                                          + "> but found " + vec.size());
595            }
596            if (max > 0 && vec.size() > max) {
597                throw new XOMException("Expecting at most " + max + " <"
598                                          + elemClass.getName()
599                                          + "> but found " +
600                                          vec.size());
601            }
602    
603            // Finally, convert to an array and return.
604            NodeDef[] retval = new NodeDef[vec.size()];
605            for (int i = 0; i < retval.length; i++) {
606                retval[i] =
607                    ElementDef.constructElement((DOMWrapper)(vec.elementAt(i)),
608                                                thisEnclosure, prefix);
609            }
610            return retval;
611        }

613        /**
614         * This function retrieves an Element from this parser, advancing the
615         * parser if the element is found.  The Element's corresponding
616         * ElementDef class is looked up and its constructor is called
617         * automatically.  If the requested Element is not found the function
618         * returns null <i>unless</i> required is set to true.  In this case,
619         * a XOMException is thrown.
620         * @param elementClass the Class of the element to retrieve.
621         * @param required true to throw an exception if the element is not
622         * found, false to simply return null.
623         * @return the element, as an ElementDef, or null if it is not found
624         * and required is false.
625         * @throws XOMException if required is true and the element could not
626         * be found.
627         */
628        public NodeDef getElement(Class elementClass,
629                                  boolean required)
630            throws XOMException
631        {
632            // If current element is null, return null immediately
633            if (currentChild == null) {
634                return null;
635            }
636    
637            // Check if the name matches the class
638            if (!nameMatchesClass(elementClass)) {
639                if (required) {
640                    throw new XOMException("element <" + currentChild.getTagName()
641                                              + "> is not of expected type "
642                                              + elementClass.getName());
643                } else {
644                    return null;
645                }
646            }
647    
648    
649    
650            // Get the class corresponding to the current tag.  This will be
651            // equal to elementClass if the current content was declared using
652            // an Element, but not if the current content was declared using
653            // a Class.
654            Class thisEnclosure = getEnclosureClass(elementClass);
655            Class currentClass = ElementDef.getElementClass(currentChild,
656                                                            thisEnclosure, prefix);
657    
658            // Get the element
659            DOMWrapper prevChild = currentChild;
660            getNextElement();
661    
662            // Construct an ElementDef of the correct class from the element
663            return ElementDef.constructElement(prevChild, currentClass);
664        }

685        /**
686         * This function retrieves a String element from this parser,
687         * advancing the parser if the element is found.
688         * If no element of the correct name is found, this function returns null,
689         * unless required is true, in which case a XOMException is thrown.
690         * @param elementName the name of the element to retrieve.
691         * @param required true to throw an exception if the element is not
692         * found, false to simply return null.
693         * @return the String value stored inside the element to retrieve, or
694         * null if no element with the given elementName could be found.
695         */
696        public String getString(String elementName, boolean required)
697            throws XOMException
698        {
699            boolean found;
700            if (required) {
701                requiredName(elementName);
702                found = true;
703            } else {
704                found = optionalName(elementName);
705            }
706            if (found) {
707                String retval = currentChild.getText().trim();
708                getNextElement();
709                return retval;
710            } else {
711                return null;
712            }
713        }

716         * This function returns a collection of String elements of the given
717         * name, returning them as an array.
718         * @param elemName the element name.
719         * @param min the minimum number of elements required in the array.  Set
720         * this parameter to 0 to indicate no minimum.
721         * @param max the maximum number of elements allowed in the array.  Set
722         * this parameter to 0 to indicate no maximum.
723         * @return a String array containing the discovered elements.
724         * @throws XOMException if there are fewer than min or more than max
725         * elements with the name <i>elemName</i>.
726         */
727        public String[] getStringArray(String elemName, int min, int max)
728            throws XOMException
729        {
730            // First, read the appropriate elements into a vector.
731            Vector vec = new Vector();
732            String augName = prefix + elemName;
733            while(currentChild != null &&
734                  augName.equalsIgnoreCase(currentChild.getTagName().toString())) {
735                vec.addElement(currentChild);
736                getNextElement();
737            }
738    
739            // Now, check for size violations
740            if (min > 0 && vec.size() < min) {
741                throw new XOMException("Expecting at least " + min + " <"
742                                          + elemName + "> but found " + vec.size());
743            }
744            if (max > 0 && vec.size() > max) {
745                throw new XOMException("Expecting at most " + max + " <"
746                                          + elemName + "> but found " +
747                                          vec.size());
748            }
749    
750            // Finally, convert to an array, retrieve the text from each
751            // element, and return.
752            String[] retval = new String[vec.size()];
753            for (int i = 0; i < retval.length; i++)
754                retval[i] = ((DOMWrapper)(vec.elementAt(i))).getText().trim();
755            return retval;
756        }

758        // Determine if a String is present anywhere in a given array.
759        private boolean stringInArray(String str, String[] array)
760        {
761            for (int i = 0; i < array.length; i++) {
762                if (str.equals(array[i])) {
763                    return true;
764                }
765            }
766            return false;
767        }

769        // Convert an array of Strings into a single String for display.
770        private String arrayToString(String[] array)
771        {
772            StringBuffer sbuf = new StringBuffer();
773            sbuf.append("{");
774            for (int i = 0; i < array.length; i++) {
775                sbuf.append(array[i]);
776                if (i < array.length - 1) {
777                    sbuf.append(", ");
778                }
779            }
780            sbuf.append("}");
781            return sbuf.toString();
782        }

784        /**
785         * Get a Class object representing a plugin class, identified either
786         * directly by a Java package and Java class name, or indirectly
787         * by a Java package and Java class which defines a method called
788         * getXMLDefClass() to return the appropriate class.
789         * @param packageName the name of the Java package containing the
790         * plugin class.
791         * @param className the name of the plugin definition class.
792         * @throws XOMException if the plugin class cannot be located
793         * or if the designated class is not suitable as a plugin class.
794         */
795        public static Class getPluginClass(String packageName,
796                                           String className)
797            throws XOMException
798        {
799            Class managerClass = null;
800            try {
801                managerClass = Class.forName(packageName + "." + className);
802            } catch (ClassNotFoundException ex) {
803                throw new XOMException("Unable to locate plugin class "
804                                          + packageName + "."
805                                          + className + ": "
806                                          + ex.getMessage());
807            }
808    
809            return getPluginClass(managerClass);
810        }

812        /**
813         * Get a Class object representing a plugin class, given a manager
814         * class that implements the static method getXMLDefClass().
815         * @param managerClass any Class that implements getXMLDefClass.
816         * @return the plugin Class.
817         */
818        public static Class getPluginClass(Class managerClass)
819            throws XOMException
820        {
821            // Look for a static method called getXMLDefClass which returns
822            // type Class.  If we find this method, call it to produce the
823            // actual plugin class.  Otherwise, throw an exception; the
824            // class we selected is inappropriate.
825            Method[] methods = managerClass.getMethods();
826            for (int i = 0; i < methods.length; i++) {
827                // Must be static, take no args, and return Class.
828                if (methods[i].getParameterTypes().length != 0) {
829                    continue;
830                }
831                if (!(methods[i].getReturnType() == Class.class)) {
832                    continue;
833                }
834                if (!(Modifier.isStatic(methods[i].getModifiers()))) {
835                    continue;
836                }
837    
838                // Invoke the method here.
839                try {
840                    Object[] args = new Object[0];
841                    return (Class)(methods[i].invoke(null, args));
842                } catch (InvocationTargetException ex) {
843                    throw new XOMException("Exception while retrieving "
844                                              + "plugin class: " +
845                                              ex.getTargetException().toString());
846                } catch (IllegalAccessException ex) {
847                    throw new XOMException("Illegal access while retrieving "
848                                              + "plugin class: " +
849                                              ex.getMessage());
850                }
851            }
852    
853            // Class is inappropriate.
854            throw new XOMException("Plugin class " + managerClass.getName()
855                                      + " is not an appropriate plugin class; "
856                                      + "getXMLDefClass() is not defined.");
857        }

859        /**
860         * Retrieve an Attribute from the parser.  The Attribute may be of any
861         * Java class, provided that the class supports a constructor from the
862         * String class.  The Attribute's value will be returned as an Object,
863         * which must then be cast to the appropraite type.  If the attribute
864         * is not defined and has no default, either null is returned (if
865         * required is false), or a XOMException is thrown (if required is
866         * true).
867         * @param attrName the name of the attribute to retreive.
868         * @param attrType a String naming a Java Class to serve as the type.
869         * If attrType contains a "." character, the class is looked up directly
870         * from the type name.  Otherwise, the class is looked up in the
871         * java.lang package.  Finally, the class must have a constructor which
872         * takes a String as an argument.
873         * @param defaultValue the default value for this attribute.  If values
874         * is set, the defaultValue must also be one of the set of values.
875         * defaultValue may be null.
876         * @param values an array of possible values for the attribute.  If
877         * this parameter is not null, then the attribute's value must be one
878         * of the listed set of values or an exception will be thrown.
879         * @param required if set, then this function will throw an exception
880         * if the attribute has no value and defaultValue is null.
881         * @return the Attribute's value as an Object.  The actual class of
882         * this object is determined by attrType.
883         */
884        public Object getAttribute(String attrName, String attrType,
885                                   String defaultValue, String[] values,
886                                   boolean required)
887            throws XOMException
888        {
889            // Retrieve the attribute type class
890            if (attrType.indexOf('.') == -1) {
891                attrType = "java.lang." + attrType;
892            }
893            Class typeClass = null;
894            try {
895                typeClass = Class.forName(attrType);
896            } catch (ClassNotFoundException ex) {
897                throw new XOMException("Class could not be found for attribute "
898                                          + "type: " + attrType + ": "
899                                          + ex.getMessage());
900            }
901    
902            // Get a constructor from the type class which takes a String as
903            // input.  If one does not exist, throw an exception.
904            Class[] classArray = new Class[1];
905            classArray[0] = java.lang.String.class;
906            Constructor stringConstructor = null;
907            try {
908                stringConstructor = typeClass.getConstructor(classArray);
909            } catch (NoSuchMethodException ex) {
910                throw new XOMException("Attribute type class " +
911                                          attrType + " does not have a "
912                                          + "constructor which takes a String: "
913                                          + ex.getMessage());
914            }
915    
916            // Get the Attribute of the given name
917            Object attrVal = wrapper.getAttribute(attrName);
918            if (attrVal == null)
919                attrVal = defaultValue;
920    
921            // Check for null
922            if (attrVal == null) {
923                if (required)
924                    throw new XOMException(
925                        "Attribute '" + attrName +
926                        "' is unset and has no default value.");
927                else
928                    return null;
929            }
930    
931            // Make sure it is on the list of acceptable values
932            if (values != null) {
933                if (!stringInArray(attrVal.toString(), values))
934                    throw new XOMException("Value '" + attrVal.toString()
935                                              + "' of attribute '"
936                                              + attrName + "' has illegal value '"
937                                              + attrVal + "'.  Legal values: "
938                                              + arrayToString(values));
939            }
940    
941            // Invoke the constructor to get the final object
942            Object[] args = new Object[1];
943            args[0] = attrVal.toString();
944            try {
945                return stringConstructor.newInstance(args);
946            } catch (InstantiationException ex) {
947                throw new XOMException("Unable to construct a " + attrType
948                                          + " from value \"" + attrVal + "\": "
949                                          + ex.getMessage());
950            } catch (InvocationTargetException ex) {
951                throw new XOMException("Unable to construct a " + attrType
952                                          + " from value \"" + attrVal + "\": "
953                                          + ex.getMessage());
954            } catch (IllegalAccessException ex) {
955                throw new XOMException("Unable to construct a " + attrType
956                                          + " from value \"" + attrVal + "\": "
957                                          + ex.getMessage());
958            }
959        }
