040        /**
041         * getElementClass is a static helper function which finds the XMLDef class
042         * corresponding to an Element.  The Element's tag must start with the
043         * given prefix name, and
044         * the remainder of the tag must correspond to a static inner class of
045         * the given enclosing class.
046         * @param wrapper the DOMWrapper whose class to look up.
047         * @param enclosure a Class which encloses the Class to lookup.
048         * @param prefix a prefix which must appear on the tag name.
049         * @return the ElementDef Class corresponding to the element, or null if no
050         * class could be found (possible if this is a String element.
051         */
052        public static Class getElementClass(DOMWrapper wrapper,
053                                            Class enclosure,
054                                            String prefix)
055            throws XOMException
056        {
057            if(enclosure == null) {
058                // don't try to find a class -- they will use GenericDef
059                return null;
060            }
061            // wrapper must be of ELEMENT type.  If not, throw a XOMException.
062            if(wrapper.getType() != DOMWrapper.ELEMENT)
063                throw new XOMException("DOMWrapper must be of ELEMENT type.");
064    
065            // Retrieve the tag name.  It must start with the prefix.
066            String tag = wrapper.getTagName();
067            if(prefix == null)
068                prefix = "";
069            else if(!tag.startsWith(prefix))
070                throw new XOMException("Element names must start "
071                                          + "\"" + prefix + "\": "
072                                          + tag + " is invalid.");
073    
074            // Remove the prefix and look for the name in the _elements field
075            // of the enclosure class.  Note that the lookup is case-sensitive
076            // even though XML tags are not.
077            String className = tag.substring(prefix.length(), tag.length());
078            className = XOMUtil.capitalize(className);
079            Class elemClass = null;
080            try {
081                elemClass = Class.forName(enclosure.getName() + "$"
082                                          + className);
083            } catch (ClassNotFoundException ex) {
084                return null;
085            }
086            return elemClass;
087        }

089        /**
090         * constructElement is a helper function which builds the appropriate type
091         * of ElementDef from an XML Element.  This version of the function takes
092         * an Element and a Class object specifying the exact class to use in
093         * constructing the element.
094         * @param wrapper the DOM Element wrapper from which to build this class.
095         * @param clazz the Class to use to construct this class.  It must have
096         * a constructor which takes the Element type.
097         * @return a fully constructed ElementDef of the type specified by
098         * Class.
099         * @throws XOMException if clazz has no constructor which takes Element,
100         * or if construction fails.
101         */
102        public static NodeDef constructElement(DOMWrapper wrapper,
103                                               Class elemClass)
104            throws XOMException
105        {
106            // Find a constructor of this class which takes an "Element" object
107            Constructor[] constructors = elemClass.getDeclaredConstructors();
108            Constructor elemConstructor = null;
109            for(int i=0; i<constructors.length; i++) {
110                Class[] params = constructors[i].getParameterTypes();
111                if(params.length == 1 && params[0] == DOMWrapper.class) {
112                    elemConstructor = constructors[i];
113                    break;
114                }
115            }
116            if(elemConstructor == null)
117                throw new XOMException("No constructor taking class DOMWrapper "
118                                          + "could be found in class "
119                                          + elemClass.getName());
120    
121            // Call the constructor to instantiate the object
122            Object[] args = new Object[1];
123            args[0] = wrapper;
124            try {
125                return (ElementDef)(elemConstructor.newInstance(args));
126            } catch (InstantiationException ex) {
127                throw new XOMException("Unable to instantiate object of class "
128                                          + elemClass.getName() + ": "
129                                          + ex.getMessage());
130            } catch (InvocationTargetException ex) {
131                // the Element constructor can only throw XOMException or
132                // RuntimeException or Error, so cast to whichever type is appropriate
133                // and throw here.
134                Throwable target = ex.getTargetException();
135                if(target instanceof XOMException)
136                    throw (XOMException)target;
137                else if(target instanceof RuntimeException)
138                    throw (RuntimeException)target;
139                else if(target instanceof Error)
140                    throw (Error)target;
141                else
142                    throw new XOMException("Unexpected exception while "
143                                              + "instantiating object: "
144                                              + target.toString());
145            } catch (IllegalAccessException ex) {
146                throw new XOMException("Unable to instantiate object of class "
147                                          + elemClass.getName() + ": "
148                                          + ex.getMessage());
149            }
150        }

152        /**
153         * constructElement is a helper function which builds the appropriate type
154         * of ElementDef from an XML Element.  This function should be used when
155         * creating an ElementDef from a list of optional XML element types using
156         * ElementParser.requiredOption.  Generally, it is better to call the
157         * constructors of ElementDef subclasses directly if the exact type of
158         * an element is known.
159         * @param wrapper the DOM Element Wrapper from which to build this class.
160         * @return an ElementDef whose exact type depends on the tag name of the
161         * element definition.
162         * @throws XOMException if no subclass of ElementDef can be found,
163         * or if def is malformed.
164         */
165        public static NodeDef constructElement(
166            DOMWrapper wrapper, Class enclosure, String prefix)
167            throws XOMException
168        {
169            switch (wrapper.getType()) {
170            case DOMWrapper.ELEMENT:
171                Class elemClass = getElementClass(wrapper, enclosure, prefix);
172                if (elemClass == null) {
173                    if (true) {
174                        return new WrapperElementDef(wrapper, enclosure, prefix);
175                    } else {
176                        throw new XOMException("No class corresponding to element "
177                                               + wrapper.getTagName()
178                                               + " could be found in enclosure "
179                                               + enclosure.getName());
180                    }
181                } else {
182                    return constructElement(wrapper, elemClass);
183                }
184            case DOMWrapper.COMMENT:
185                return new CommentDef(wrapper.getText());
186            case DOMWrapper.CDATA:
187                return new CdataDef(wrapper.getText());
188            case DOMWrapper.FREETEXT:
189                return new TextDef(wrapper.getText());
190            default:
191                throw new XOMException("Unknown type: " + wrapper.getText());
192            }
193        }

421        /**
422         * This convenience function displays differences in two versions of
423         * the same string object.
424         * @param name the object name.
425         * @param value1 the first string.
426         * @param value2 the second string.
427         * @param out the PrintWriter to which to write differences.
428         * @param indent the indentation level.
429         * @return true if the strings match, false if not.
430         */
431        protected static boolean displayStringDiff(String name,
432                                                   String value1, String value2,
433                                                   PrintWriter out, int indent)
434        {
435            // True if both values are null.
436            if(value1 == null && value2 == null)
437                return true;
438    
439            // Deal with the cases where one value is set but the other is not.
440            if(value2 == null) {
441                if(out != null) {
442                    displayIndent(out, indent);
443                    out.println("String " + name + ": mismatch: "
444                                + value1.toString() + " vs null.");
445                }
446                return false;
447            }
448            if(value1 == null) {
449                if(out != null) {
450                    displayIndent(out, indent);
451                    out.println("String " + name + ": mismatch: "
452                                + "null vs " + value2.toString() + ".");
453                }
454                return false;
455            }
456    
457            // Finally, check the values themselves
458            if(value1.equals(value2))
459                return true;
460    
461            if(out != null) {
462                displayIndent(out, indent);
463                out.println("String " + name + ": mismatch: "
464                            + value1.toString() + " vs "
465                            + value2.toString() + ".");
466            }
467            return false;
468        }

470        /**
471         * This convenience function displays differences in two versions of
472         * the same XML attribute value.
473         * @param name the attribute name.
474         * @param value1 the first attribute value.
475         * @param value2 the second attribute value.
476         * @param out the PrintWriter to which to write differences.
477         * @param indent the indentation level.
478         * @return true if the values match, false if not.
479         */
480        protected static boolean displayAttributeDiff(String name,
481                                                      Object value1, Object value2,
482                                                      PrintWriter out, int indent)
483        {
484            // True if both values are null.
485            if(value1 == null && value2 == null)
486                return true;
487    
488            // Deal with the cases where one value is set but the other is not.
489            if(value2 == null) {
490                if(out != null) {
491                    displayIndent(out, indent);
492                    out.println("Attribute " + name + ": mismatch: "
493                                + value1.toString() + " vs null.");
494                }
495                return false;
496            }
497            if(value1 == null) {
498                if(out != null) {
499                    displayIndent(out, indent);
500                    out.println("Attribute " + name + ": mismatch: "
501                                + "null vs " + value2.toString() + ".");
502                }
503                return false;
504            }
505    
506            // Verify that types match
507            if (value1.getClass() != value2.getClass()) {
508                if(out != null) {
509                    displayIndent(out, indent);
510                    out.println("Attribute " + name + ": class mismatch: "
511                                + value1.getClass().getName()
512                                + " vs "
513                                + value2.getClass().getName()
514                                + ".");
515                }
516                return false;
517            }
518    
519            // Finally, check the values themselves
520            if(value1.equals(value2))
521                return true;
522    
523            if(out != null) {
524                displayIndent(out, indent);
525                out.println("Attribute " + name + ": mismatch: "
526                            + value1.toString() + " vs "
527                            + value2.toString() + ".");
528            }
529            return false;
530        }

532        /**
533         * This convenience function displays differences in the values of any
534         * two ElementDefs, returning true if they match and false if not.
535         * @param name the object name.
536         * @param value1 the first value.
537         * @param value2 the second value.
538         * @param out the PrintWriter to which to write differences.
539         * @param indent the indentation level.
540         * @return true if the values match, false if not.
541         */
542        protected static boolean displayElementDiff(String name,
543                                                    NodeDef value1,
544                                                    NodeDef value2,
545                                                    PrintWriter out, int indent)
546        {
547            // True if both values are null.
548            if(value1 == null && value2 == null)
549                return true;
550    
551            // Deal with the cases where one value is set but the other is not.
552            if(value2 == null) {
553                if(out != null) {
554                    displayIndent(out, indent);
555                    out.println("Object " + name + ": mismatch: "
556                                + "(...) vs null.");
557                }
558                return false;
559            }
560            if(value1 == null) {
561                if(out != null) {
562                    displayIndent(out, indent);
563                    out.println("Object " + name + ": mismatch: "
564                                + "null vs (...).");
565                }
566                return false;
567            }
568    
569            // Verify that types match
570            if (value1.getClass() != value2.getClass()) {
571                if(out != null) {
572                    displayIndent(out, indent);
573                    out.println("Object " + name + ": class mismatch: "
574                                + value1.getClass().getName()
575                                + " vs "
576                                + value2.getClass().getName()
577                                + ".");
578                }
579                return false;
580            }
581    
582            // Do a sub equality check
583            return ((ElementDef) value1).displayDiff(
584                (ElementDef) value2, out, indent);
585        }

587        /**
588         * This convenience function diffs any array of ElementDef values with
589         * the given array name.  All differences are written to the given
590         * PrintWriter at the given indentation level.
591         * @param name the array name.
592         * @param values1 the first array.
593         * @param values2 the second array.
594         * @param out the PrintWriter to which to write differences.
595         * @param indent the indentation level.
596         * @return true if the both arrays match, false if there are any differences.
597         */
598        protected static boolean displayElementArrayDiff(String name,
599                                                         NodeDef[] values1,
600                                                         NodeDef[] values2,
601                                                         PrintWriter out, int indent)
602        {
603            int length1 = 0;
604            int length2 = 0;
605            if (values1 != null)
606                length1 = values1.length;
607            if (values2 != null)
608                length2 = values2.length;
609            // Check array sizes
610            //  a null array does not differ from an empty array
611            if(length1 != length2) {
612                if(out != null) {
613                    displayIndent(out, indent);
614                    out.println("Array " + name + ": size mismatch: "
615                                + length1 + " vs "
616                                + length2 + ".");
617                }
618                return false;
619            }
620    
621            // Check each member of the array
622            boolean diff = true;
623            for(int i=0; i< length1; i++)
624                diff = diff && displayElementDiff(name + "[" + i + "]",
625                                                  values1[i], values2[i],
626                                                  out, indent);
627            return diff;
628        }

707        /**
708         * The diff function compares this element against another, determining if
709         * they are exactly equal.  If so, the function returns null.  If not,
710         * it returns a String describing the differences.
711         */
712        public String diff(ElementDef other)
713        {
714            StringWriter writer = new StringWriter();
715            PrintWriter out = new PrintWriter(writer);

782        /**
783         * Public version of clone(); returns a deep copy of this ElementDef.
784         */
785        public ElementDef deepCopy()
786            throws XOMException
787        {
788            try {
789                ByteArrayOutputStream byteBuffer = new ByteArrayOutputStream();
790                ObjectOutputStream objOut = new ObjectOutputStream(byteBuffer);
791                objOut.writeObject(this);
792                objOut.flush();
793                ByteArrayInputStream byteIn = new
794                    ByteArrayInputStream(byteBuffer.toByteArray());
795                ObjectInputStream objIn = new ObjectInputStream(byteIn);
796                ElementDef def = (ElementDef)(objIn.readObject());
797                return def;
798            } catch (IOException ex) {
799                throw new XOMException(ex, "Failed to serialize-copy ElementDef");
800            } catch (ClassNotFoundException ex) {
801                throw new XOMException(ex, "Failed to serialize-copy ElementDef");
802            }
803        }
